		DEVICE ZXSPECTRUM128
		ORG	#8000
STARTFORSNA
MAIN	EI
		CALL	INIT_GEN
		LD	A,4
		LD	(X_TEXT_COORD),A
		LD	A,0
		LD	(Y_TEXT_COORD),A
		LD  HL,COPY_R
		LD	DE,TEXT
		CALL STRCPY
		CALL DRAW_TEXT
		LD	A,16
		LD	HL,#5800
		LD	(HL),A
		CALL	GENERATE
		CALL DRAW_COLOR_LABELS
		LD	HL,#5800
		LD	A,32
		LD	(HL),A
LOOP1
		HALT
		LD	A,#7F
		IN	A,(#FE)
		RRA
		JP	C,LOOP1
		RET

STRCPY	; HL - SOURCE, DE - DEST, ZERO ENDING
STRC1
	LD	A,(HL)
	LD	(DE),A
	INC	HL
	INC DE
	AND	A
	JR	NZ,STRC1
	RET
			
INIT_GEN
	LD	A,0
	LD	(N),A  ; [N]=0
	
	LD	A,0
	LD	(X_COORD),A
	LD	(Y_COORD),A
	
	LD	HL,VERT
	LD	DE,VERT+1
	LD	BC,7
	LD	A,1
	LD	(HL),A
	LDIR		; INIT VERT WITH 1
	
	LD	HL,VOSH
	LD	DE,VOSH+1
	LD	BC,14
	LD	A,1
	LD	(HL),A
	LDIR		; INIT VOSH WITH 1
	
	LD	HL,NISH
	LD	DE,NISH+1
	LD	BC,14
	LD	A,1
	LD	(HL),A
	LDIR		; INIT NISH	WITH 1
	
	LD	HL,RESH
	LD	DE,RESH+1
	LD	BC,7
	LD	A,0
	LD	(HL),A  ; INIT RESH WITH 0
	LDIR
	RET
	
GENERATE
	LD	A,0		
	PUSH	AF	; LOCAL VAR [H]
	
GEN_LOOP
	LD	IX,0
	ADD	IX,SP
	CALL	IF_FREE ; RETURN A
	JR	Z,NEXT_H
	
	LD	IY,RESH
	LD	A,(N)
	LD	(X_N+2),A
	LD	A,(IX+1)
X_N	LD	(IY+0),A	; X[N]=[H]
	
	LD	C,0
	CALL	SET_UNSET_FERS
	
	LD	A,(N)
	INC	A
	LD	(N),A ; [N]=[N]+1
	
	CP	8
	JR	NZ,NEXT_GEN
	;CALL  DRAW_RESHENIE
	CALL DRAW_TEXT_RESH
	LD	A,(RESULT_NUM) 
	INC	A
	LD	(RESULT_NUM),A ; COUNT NUM OF RESHENII
	CALL DRAW_NUM
	JR	CONTIN
NEXT_GEN
	CALL	GENERATE
	LD	IX,0    ; UPDATE IX
	ADD	IX,SP 
CONTIN
	LD	A,(N)
	DEC	A
	LD	(N),A ; [N]=[N]-1
	LD	C,1
	CALL	SET_UNSET_FERS	
NEXT_H
	POP	AF
	INC	A
	PUSH	AF ; [H]=[H]+1
	CP 8
	JR	NZ,GEN_LOOP
	POP		AF
	RET

IF_FREE
	LD	A,(IX+1)
	LD	HL,VERT
	LD	B,0
	LD	C,A
	ADD	HL,BC
	LD	A,(HL)
	LD	D,A

	LD	A,(IX+1) 
	LD	B,A
	LD	A,(N)
	ADD	A,7
	SUB	B			; A=[N]-[H]
	LD	HL,VOSH
	LD	B,0
	LD	C,A
	ADD	HL,BC
	LD	A,(HL)
	AND	D
	LD	D,A
	
	LD	A,(IX+1)
	LD	B,A
	LD	A,(N)
	ADD	B			; A=[N]+[H]
	LD	HL,NISH
	LD	B,0
	LD	C,A
	ADD	HL,BC
	LD	A,(HL)
	AND	D		; A=VERT[H] OR VOSH[N-H] OR NISH[N+H]
	RET
		
SET_UNSET_FERS	
	LD	IY,VERT
	LD	A,(IX+1)
	LD	(VERT_H+2),A
	LD	A,C
VERT_H	LD	(IY+0),A	; VERT[H]=C

	LD	IY,VOSH
	LD	A,(IX+1)
	LD	B,A
	LD	A,(N)
	ADD	A,7
	SUB	B
	LD	(VOSH_N_H+2),A
	LD	A,C
VOSH_N_H	LD	(IY+0),A	; VOSH[N-H]=C
	
	LD	IY,NISH
	LD	A,(IX+1)
	LD	B,A
	LD	A,(N)
	ADD	A,B
	LD	(NISH_N_H+2),A
	LD	A,C
NISH_N_H	LD	(IY+0),A	; NISH[N+H]=C
	RET

DOWN	INC	D
		LD	A,D
		AND	7
		RET	NZ
		LD	A,E
		ADD	A,32
		LD	E,A
		RET	C
		LD	A,D
		SUB	8
		LD	D,A
		RET

DRAW_TEXT_RESH
	LD	HL,RESH
	LD	DE,TEXT
	LD	B,8
DRAW_T_R
	LD	A,(HL)
	ADD	A,'1'
	LD	(DE),A
	INC	HL
	INC	DE
	DJNZ	DRAW_T_R
	LD	A,0
	LD	(DE),A  ; ZERO ENDING
	
	CALL	CALC_RESH_POS
	CALL	DRAW_TEXT
	RET

DRAW_COLOR_LABELS
	LD	HL,#5820
	LD	DE,8
	LD	A,16+32
	LD	B,92
DRAW_C_L
	LD	(HL),A
	ADD	HL,DE
	DJNZ	DRAW_C_L
	RET
	
CALC_RESH_POS
	LD	A,(RESULT_NUM)
	SRL	A
	SRL	A
	INC	A
	LD	(Y_TEXT_COORD),A
	LD	A,(RESULT_NUM)
	AND	3
	SLA	A
	SLA	A
	SLA	A
	LD	(X_TEXT_COORD),A
	RET
		
DRAW_TEXT
	LD	A,(Y_TEXT_COORD)
	LD	C,A
	LD	A,(X_TEXT_COORD)
	LD	B,A
	CALL CALCADDRPOS
	LD	DE,TEXT
	EX	DE,HL
DT1	
	LD	A,(HL)
	AND A
	RET Z
	PUSH	HL
	PUSH	DE
	CALL	DRAW_SYMBOL
	POP		DE
	POP		HL
	INC		DE
	INC		HL
	JR	DT1
	RET

DRAW_SYMBOL  ; A - SYMBOL, DE - SCREEN ADDR 
		LD	L,A
		LD	H,0
		ADD	HL,HL
		ADD HL,HL
		ADD HL,HL
		PUSH	DE
		LD	DE,(23606)
		ADD	HL,DE
		POP	DE
		LD	B,8
DRAW_SYMB_L
		LD	A,(HL)
		LD	(DE),A
		INC	HL
		CALL DOWN
		DJNZ	DRAW_SYMB_L
	RET

DRAW_NUM
	LD	A,(RESULT_NUM)
	LD	B,10
	LD	C,0
DRAW_NUM1
	INC	C
	SUB	B
	JR	NC,DRAW_NUM1
	DEC C
	ADD	A,B  
	ADD	A,'0'
	LD	(TEXT+1),A
	LD	A,C
	ADD	A,'0'
	LD	(TEXT),A
	LD	A,0
	LD	(TEXT+2),A
	LD	A,1
	LD	(X_TEXT_COORD),A
	LD	A,0
	LD	(Y_TEXT_COORD),A
	CALL	DRAW_TEXT
	RET
	
CALCADDRPOS ; B - x, C - y char place, return addr in HL
	PUSH DE
	LD	HL,#4000
	LD  DE,#800
	LD  A,C
	AND 8
	JR Z,SAV1
	ADD HL,DE
SAV1 LD A,C
	AND 16
	JR Z,SAV2
	ADD HL,DE
	ADD HL,DE
SAV2 LD A,C
	AND 7
	EX DE,HL
	LD H,0
	LD L,A
	ADD HL,HL
	ADD HL,HL
	ADD HL,HL
	ADD HL,HL
	ADD HL,HL
	EX DE,HL
	ADD HL,DE
	LD  D,0
	LD  E,B
	ADD HL,DE
	POP DE
	RET

SAVE_RESULT
	LD	HL,(PTR_RESULTS)
	LD	IX,RESH
	LD	B,4
SAVR1
	LD	A,(IX+0)
	SLA	A
	SLA	A
	SLA	A
	SLA	A
	LD	C,A
	LD	A,(IX+1)
	OR	C
	LD	(HL),A
	INC	HL
	INC IX
	INC IX
	DJNZ	SAVR1
	LD	(PTR_RESULTS),HL
	RET

SEARCH_RESULT
	LD	BC,4
SEARCH
	
	RET

MIRROR_X
	LD	B,8
	LD	C,3
	LD	HL,RESH
M_X	
	LD	A,(HL)
	CP	4
	JR	C,M_X1
	SUB	3
	LD	C,A
	LD	A,4
	SUB C
	JR	M_X2
M_X1
	LD	C,A
	LD	A,3
	SUB	C
	ADD	4
M_X2
	INC	HL
	DJNZ	M_X
		RET

X_COORD	DEFB	0
Y_COORD DEFB	0

X_TEXT_COORD	DEFB 0
Y_TEXT_COORD	DEFB 0
TEXT	DEFS	33
COPY_R	DEFB	"EIGHT QUEENS DIJKSTRA @CSG17",0

N		DEFB	0
RESH	DEFS	8
VERT	DEFS	8
VOSH	DEFS	15
NISH	DEFS	15

RESULT_NUM	DEFB 0

PTR_RESULTS	DEFW	RESULTS_ALL
RESULTS_ALL	DEFS	92*4
RESULTS_EXC	DEFS	12*4
SEARCH_PAT	DEFS	4

 SAVESNA "8queen.sna",STARTFORSNA
 SAVEBIN "8queen.bin",STARTFORSNA,$-STARTFORSNA
